Table employees {
  employee_id int [pk, increment]
  login varchar(20) [unique, not null]
  password_hash varchar(256) [not null]
  email varchar(255) [not null]
  last_name varchar(50) [not null]
  first_name varchar(50) [not null]
  patronymic varchar(50) [not null]
  xp int [default: 0, not null]
  level int [default: 1, not null]
  money int [default: 0, not null]
}

Table tasks {
  task_id int [pk, increment]
  title varchar(20) [not null]
  description varchar(50) [not null]
  xp_reward int [not null]
  created_at datetime [default: 1]
}

Table achievements {
  achivement_id int [pk, increment]
  title varchar(20) [not null]
  description varchar(50) [not null]
  xp_reward int [not null]
}

Table teams {
  team_id int [pk, increment]
  title varchar(20) [not null]
  created_at datetime [not null]
}

Table team_employee {
  team_id int [ref: > teams.team_id]
  employee_id int [ref: > employees.employee_id]
}

Table achievement_employee {
  achievement_id int [ref: > achievements.achivement_id]
  employee_id int [ref: > employees.employee_id]
  received_at datetime [not null]
}

Table task_employee {
  task_id int [ref: > tasks.task_id]
  employee_id int [ref: > employees.employee_id]
  took_at datetime [not null]
  completed_at datetime [not null]
}
